eval() 함수와 JSON 문자열 연동 

/*
	eval() 함수와 JSON 문자열 연동
 	JSON : 색인배열 => eval() OK
 		   연관배열 => X why eval("{key:값}")
 		               O eval("({key:값})")
	연관배열 문자열을 eval()함수 사용하여 객체화 하는 방법
*/

var str = '{name : "이순신",age : 30}';
var obj1 = eval("(" + str + ")" );

document.write(obj1.name + " : " + obj1.age);
