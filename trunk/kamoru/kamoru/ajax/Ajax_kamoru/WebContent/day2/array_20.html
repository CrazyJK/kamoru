색인배열 문자열을 eval()함수 사용하여 객체화 하는 방법 

var str = '[' +
   		'{name : "이순신", age : 30},' +
   		'{name : "이순신2", age : 32},' +
   		'{name : "이순신3", age : 33},' +
   		'{name : "이순신4", age : 34}' + 
   ']';
var arr = eval(str);

for(var i=0 ; i<arr.length ; i++){
	document.write(arr[i].name + " : " + arr[i].age + "
");
}
