/*

   JSON을 사용하여 여러건을 객체화한 경우
 */

//객체 생성 색인배열[]로 연관배열{}을 묶는다
var arr = [
   		{name : "이순신", age : 30},
		{name : "이순신2", age : 32},
		{name : "이순신3", age : 33},
		{name : "이순신4", age : 34}
];


for(var i=0 ; i < arr.length ; i++){
	document.write(arr[i].name + " : " + arr[i].age + "
");
}
